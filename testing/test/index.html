def normalized_bump(x, n: int):
    # A_n = 2^n (n!)^2 / (2n)!
    logA = n * jnp.log(2.0) + 2.0 * gammaln(n + 1.0) - gammaln(2.0 * n + 1.0)
    A = jnp.exp(logA)
    return A * (1.0 - jnp.cos(2.0 * jnp.pi * x)) ** n
